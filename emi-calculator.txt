        <input type="number" id="rate" placeholder="Enter annual interest rate">
        
        <label for="tenure">Tenure (months):</label>
        <input type="number" id="tenure" placeholder="Enter tenure in months">
        
        <button onclick="calculateEMI()">Calculate EMI</button>
        
        <div class="result" id="emiResult"></div>
    </div>

    <script>
        let currencySymbol = "Rs";
        
        function calculateEMI() {
            let P = document.getElementById("amount").value;
            let annualRate = document.getElementById("rate").value;
            let n = document.getElementById("tenure").value;
            
            let r = annualRate / 12 / 100;
            let EMI = (P * r * Math.pow(1 + r, n)) / (Math.pow(1 + r, n) - 1);
            
            if (!isNaN(EMI) && EMI !== Infinity) {
                document.getElementById("emiResult").innerHTML = `Monthly EMI: ${currencySymbol} ${EMI.toFixed(2)}`;
            } else {
                document.getElementById("emiResult").innerHTML = "Please enter valid inputs";
            }
        }
    </script>
